01-Vue基础入门

Vue目录结构

目录/文件 说明
build 项目构建(webpack)相关代码
config 配置目录，包括端口号等。我们初学可以使用默认的。
node_modules npm 加载的项目依赖模块
src 这里是我们要开发的目录，基本上要做的事情都在这个目录里。里面包含了几个目录及文件：assets: 放置一些图片，如logo等。components: 目录里面放了一个组件文件，可以不用。App.vue: 项目入口文件，我们也可以直接将组件写这里，而不使用 components 目录。main.js: 项目的核心文件。
static 静态资源目录，如图片、字体等。
test 初始测试目录，可删除
.xxxx文件 这些是一些配置文件，包括语法配置，git配置等。
index.html 首页入口文件，你可以添加一些 meta 信息或统计代码啥的。
package.json 项目配置文件。
README.md 项目的说明文档，markdown 格式

computed计算属性

<body>
    <div id="app">
        <p>总价格:{{totalPrice}}</p>
        <p>单价:{{price}}</p>
        <p>数量:{{num}}</p>
        <div>
            <!-- num可以直接使用不加this -->
            <button @click="num==0?0:num--">减少数量</button>
            <button @click="num++">增加数量</button>
            <button @click="logTotalPrice">打印totalPrice</button>
        </div>
    </div>

    <script>
        var vue_data={
            price:20,
            num:0,
        }
        var vm=new Vue({
            el:'#app',
            data:vue_data,
            computed:{
                // 计算属性类似于方法，但不是方法
                totalPrice(){
                    console.log('totalPrice计算属性被执行计算了')
                    return this.price*this.num
                }
            },
            methods:{
                logTotalPrice(){
                    console.log('totalPrice的结果'+this.totalPrice)
                }
            }
        })
        
    </script>
</body>

filters过滤器

<body>
    <div id="app">
        <!-- 使用 | 触发过滤器，把msg的值传递给toUppercase过滤器 -->
        <p>{{msg | toUppercase}}</p>
    </div>

    <script>
        var data_obj={
            msg:'helloworld'
        }
        var vm=new Vue({
            el:'#app',
            data:data_obj,
            filters:{
                toUppercase(value){
                    // 过滤器先判断 传过来的值是否位空
                    return value?value.toUpperCase():''
                }
            }
        })
    </script>
</body>

扩展：对html标签属性使用过滤器

<body>
    <div id="app">
        <!-- 过滤出正确的id -->
        <p v-bind:id="dataId|formatId">helloworld</p>
        {{dataId}}
    </div>

    <script>
        var data_obj = {
            dataId:'dff1'
        }
        var vm = new Vue({
            el: '#app',
            data: data_obj,
            filters: {
                formatId(value) {
                    // f + 0
                    return value ? value.charAt(1) + value.indexOf('d') : ''
                }
            }
        })
    </script>
</body>

组件传递-子传父$emit

<body>
    <div id="app">
        <parent></parent>
    </div>
    <template id='parent'>
        <!-- template内部 需要一个根元素div包裹住 -->
        <!-- 因为组件(<parent>)代表的是一个组件(一个根元素) -->
        <div>
            <!-- 父组件里是 子组件 加 p标签 -->

            <!-- 组件里调用的方法是所处div的vue组件对象的方法，这里是parent的方法 -->
            <child @childfn='change'></child>
            <p>子组件传递的值: {{msg}} </p>
        </div>
    </template>
    <!-- 子组件  -->
    <template id='child'>
        <div>
            <button @click='send'>Send</button>
            <input type="text" v-model='cmsg'>
        </div>
    </template>
    <script>
        Vue.component('parent',{
            template:'#parent',
            data(){
                return {
                    msg:''
                }
            },
            methods:{
                //父组件方法，由子组件的自定义事件触发此方法，从而接收子组件传递的值
                change(msg,age){
                    this.msg=msg+age
                }
            }
        })
        Vue.component('child',{
            template:'#child',
            data(){
                return {
                    cmsg:'子组件的消息'
                }
            },
            methods:{
                //子组件方法
                send(){
                    // 通过$emit调用自定义事件，然后传递数据(,携带的参数从子组件传给了父组件)
                    this.$emit('childfn',this.cmsg,18)
                    // childfn -> click2
                }
            }
        })

        var vm=new Vue({
            el:'#app'
        })
    </script>
</body>

                {path:'/user/:name',name:'user',component:user}
            ]
        })
        var vm=new Vue({
            el:'#app',
            methods:{
                goStart(){
                    this.$router.push({
                        // 命名路由的方式 
                        name:'user', //指定要跳转的路由名称
                        params:{name:'admin'} //携带的参数
                    })
                },
                goBack(){
                    //this.$router.go(-1)相当于history.back()
                    //this.$router.go(1)相当于history.forword()
                    this.$router.go(-1) // 进行后退操作
                }
            },
            router
        })
    </script>
